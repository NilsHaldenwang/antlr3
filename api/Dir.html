  entries.map! do |entry|

def self.mkpath(path)
  $VERBOSE and $stderr.puts("INFO: Dir.mkpath(%p)" % path)
  test(?d, path) and return(path)
  parent = File.dirname(path)
  test(?d, parent) or mkpath(parent)
  Dir.mkdir(path)
  return(path)
end
