def initialize( options = {} )
  @listener = listener
  @host = options.fetch( :host, 'localhost' )
  @port = options.fetch( :port, DEFAULT_PORT )
  @buffer = StringIO.new
  super do
    connect do
      handshake
      loop do
        yield( read_event )
      end
    end
  end
end

def parse_string( string )
  @buffer.string = string
  @buffer.rewind
  out = ''
  until @buffer.eof?
    case c = @buffer.getc
    when ?\\                    # escape
      nc = @buffer.getc
      out << 
        if nc.between?( ?0, ?9 )  # octal integer
          @buffer.ungetc( nc )
          @buffer.read( 3 ).to_i( 8 ).chr
        elsif nc == ?x
          @buffer.read( 2 ).to_i( 16 ).chr
        else
          ESCAPE_MAP[ nc ]
        end
    else
      out << c
    end
  end
  return out
end

def unpack_event(event)
  event.nil? and raise( StopIteration )
  event.chomp!
  name, *elements = event.split("\t",-1)
  name = name.to_sym
  name == :terminate and raise StopIteration
  elements.map! do |elem|
    elem.empty? and next(nil)
    case elem
    when /^\d+$/ then Integer(elem)
    when /^\d+\.\d+$/ then Float(elem)
    when /^true$/ then true
    when /^false$/ then false
    when /^"(.*)"$/ then parse_string($1)
    end
  end
  elements.unshift(name)
  return( elements )
end
