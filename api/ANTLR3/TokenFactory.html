def token_class
  @token_class ||= begin
    self.class.token_class rescue
    self::Token rescue
    ANTLR3::CommonToken
  end
end

def create_token( *args )
  if block_given?
    token_class.new( *args ) do |*targs|
      yield( *targs )
    end
  else
    token_class.new( *args )
  end
end
