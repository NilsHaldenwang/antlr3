def self.parse(call_string)
  parts = call_string.split(':', 3)
  file = parts.shift
  line = parts.shift.to_i
  if parts.empty?
      return Call.new(file, line)
  else
      mstring = parts.shift
      match = mstring.match(/`(.+)'/)
      method = match ? match[1] : nil
      return Call.new(file, line, method)
  end
end
