def message
  if @inserted and @symbol
    "%s: inserted %p at %p" %
      [ self.class, @inserted, @symbol.text ]
  else
    msg = self.class.to_s
    msg << ': at %p' % token.text unless @token.nil?
    return msg
  end
end
